// 11. Allow only 3 attempts to enter correct password
// If user gets it right early, stop. If not → “Account locked”

let pass = "mannat"
let attempts = 0
let password = prompt("enter your password")
attempts++

while(pass !== password && attempts < 3){
  if (password === null) {
    console.error("Operation cancelled by user!");
    break;
  }

  if (password.trim() === "") {
    console.error("Password cannot be empty!");
    break
  }
  password = prompt("enter your password")
  attempts++
}
console.log(`Total attempts required: ${attempts}`)

// 15. Keep asking number until user enters an even number
// Use while loop. Stop only if input is even.
let tries = 0;
let num = prompt("Enter a number");

while (true) {
  if (num === null || num.trim() === "") {
    console.error("Enter something!");
    break;
  }

  num = Number(num);
  tries++;

  if (isNaN(num)) {
    console.error("Please enter a valid number!");
    num = prompt("Enter again (need even number)");
    continue;
  }

  if (num % 2 === 0) {
    console.log("Nice! Even number entered.");
    break;
  }

  num = prompt("Enter again (need even number)");
}

console.log(`Total tries: ${tries}`);

// 19. ATM Simulator – Allow 3 withdrawals
// Start with ₹1000 balance. Ask withdrawal amount 3 times.
// If enough balance → deduct
// Else → print “Insufficient balance”

let hi = 0
let balance = 1000 

while(hi !== 3){
    let withdrawal = +prompt(`enter withdrawal amount from ${balance}`)

    if(balance < withdrawal){
        console.error("Insufficient balance")
        break
    }

    balance -= withdrawal
    hi++
}
console.log(`remaining balance is ${balance}`)
